import random

# Predefined word list
word_list = ['apple', 'banana', 'cherry', 'grape', 'mango']
secret_word = random.choice(word_list)

# Game setup
guessed_letters = []
tries_left = 6
display_word = ['_'] * len(secret_word)

print("ğŸ® Welcome to Hangman!")
print("Guess the word, one letter at a time.")
print("You have", tries_left, "wrong attempts allowed.")
print(" ".join(display_word))

# Game loop
while tries_left > 0 and '_' in display_word:
    guess = input("\nEnter a letter: ").lower()

    if len(guess) != 1 or not guess.isalpha():
        print("â— Please enter a single alphabet letter.")
        continue

    if guess in guessed_letters:
        print("âš ï¸ You already guessed that letter.")
        continue

    guessed_letters.append(guess)

    if guess in secret_word:
        print("âœ… Good guess!")
        for i in range(len(secret_word)):
            if secret_word[i] == guess:
                display_word[i] = guess
    else:
        tries_left -= 1
        print("âŒ Wrong guess! Tries left:", tries_left)

    print("Word: " + " ".join(display_word))
    print("Guessed letters: " + ", ".join(guessed_letters))

# Game result
if '_' not in display_word:
    print("\nğŸ‰ Congratulations! You guessed the word:", secret_word)
else:
    print("\nğŸ’€ Game over! The word was:", secret_word)
